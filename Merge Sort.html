Merge Sort

Merge Sort is an example of the divide and conquer algorithmic paradigm. The algorithm was invented by John von Neumann in 1945. Merge sort divides the array of elements into smaller subarrays and sort them and then merge these to form a complete sorted array. Conceptually, merge sort works as follows:

Divide the unsorted list into two sublists of about half the size.
Divide each of the two sublists continuously until we have list sizes of length 1, in which case the list itself is returned.
Merge the two sorted sublists back into one sorted list.

The merge sort algorithm:

  1. 1. Divide the list in half
  2. 2. Merge sort the first half
  3. 3. Merge sort the second half
  4. 4. Merge both halves back together

Merge Sort Algorithm

public class mergesort {

         void sort(int[] randomNumbersArray)
         {
                  int []copy = randomNumbersArray.clone();
                  mergeSort( 0 , copy.length - 1, copy);
         }
         void mergeSort(int low, int high, int copy[])
         {
                  if( low < high)
                  {
                          int mid = (( low + high) / 2);
                          mergeSort( low, mid, copy); //left side
                          mergeSort( mid + 1, high, copy); // right side
                          merge( low, mid, high, copy); //combine them
                  }
         }

         void merge( int low, int mid, int high, int copy[])
         {
         int temp[] = new int[high - low + 1];
         int left = low;
         int right = mid + 1;
         int index = 0;
         // compare each item for equality
         while(left <= mid && right <= high)
         {
                  if ( copy[left] < copy[right])
                  {
                          temp[index] = copy[left];
                          left++;
                  }else
                  {
                          temp[index] = copy[right];
                          right++;
                  }
                  index++;
         }
         // if there is any remaining loop through them
         while(left <= mid || right <= high)
         {
                  if (left <= mid)
                  {
                          temp[index] = copy[left];
                          left++;
                          index++;
                  } else if(right <= high)
                  {
                          temp[index] = copy[right];
                          right++;
                          index++;
                  }
         }
         // copy back to array
         for(int i = 0; i < temp.length; i++)
         {
                          copy[low + i] = temp[i];
         }
         }

}
